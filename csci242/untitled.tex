        \item $f(n) = 2*n + n^2 + 5n^3 \in$ \framebox{$8*O(n^3) \forall n \geq 1$}

        \item $f(n) = n^{10} - 4 \in$ \framebox{$O(n^{10}) \forall n$}
    \end{enumerate}
    \item For each of the following $f(n)$ derive the largest Big-Omega $\Omega(g(n))$ inequality and state whether it will be true for all values of $n$.
    \begin{enumerate}
        \item $f(n) = n^3 + 25 \in$ \framebox{$\Omega(n^3) \forall n$}
        \item $f(n) = 5n \in$ \framebox{$\Omega(n) \forall n \geq 0$}
        \item $f(n) = 100n + n^2 \in$ \framebox{$\Omega(n^2) \forall n \geq 0$}

        \item $f(n) = n^2 + 3 \in$ \framebox{$\Omega(n^2) \forall n$}
        \item $f(n) = n\log{n} \in$ \framebox{$\Omega(n\log{n}) \forall n$}
    \end{enumerate}
