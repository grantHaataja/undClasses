		\item $f(n) = 8 + n^3 + 25n \leq 34 * n^3$ $\forall n \geq 1$

	\begin{enumerate}
		\item $f(n) = n^3 + 2 \geq 1 * n^3$ $\forall n$ \framebox{$f(n) = \Omega(n^3)$}
		\item $f(n) = 5n \geq 1 * n$ $\forall n$ \framebox{$f(n) = \Omega(n)$}

		\item $n^2 + 3 \geq 1 * n^2$ $\forall n$ \framebox{$f(n) = \Omega(n^2)$}
		\item $n\log n \geq 1 * n\log n$ $ \forall n$ \framebox{$f(n) = \Omega(n\log n)$}
	\end{enumerate}
\end{enumerate}
